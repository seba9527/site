# 循环结构

    a

    b

while 和 for可以带else子句，如果循环条件表达式不成立，就执行else中的语句，break语句导致的结束不执行else语句。

#求100以内的素数
for n in xrange(1,100):
    for i in xrange(2,n):
        if n%i==0:
            break
    else:
        print n
#求1到100数之和
for i in xrange(1,101):
    s+=i
print s

#求组合数C(n，i)

def Cni(n,i):
    if not (isinstance(n,int) and isinstance(i,int) and n>=i ):
        print "n and i must be integers"
        return
    result=1
    Min,Max=min(i,n-1),min(i,n-i)
    for i in xrange(n,0,-1):
        if i>Max:
            result*=i
        elif i<=Min:
            result/=i
    return result
